#include <stdio.h>
#include <stdlib.h>

int comparar (const void * a, const void * b)
{
    return ( *(int*)a - *(int*)b );
}

int valor[ 6 ] = { 50, 20, 60, 40, 10, 30 };

int main ()
{
    qsort (valor, 6, sizeof (int), comparar);

    for( int i = 0; i < 6; ++ i )
    {
        printf( "%d - %d\n", i, valor[ i ] );
    }

    return 0;
}
