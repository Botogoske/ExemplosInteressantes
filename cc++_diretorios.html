#include <stdio.h>
#include <stdlib.h>
#include <io.h>
#include <time.h>

int main()
{
    struct _finddata_t retorno;

    int handle = _findfirst( "*.*", &retorno );

    printf( "rc: %d\n\n", handle );

    do
    {
        struct tm horario;

        localtime_s( &horario, &retorno.time_create );
        printf( "name: %s, area: %d\nattributo: 0x%x\n", retorno.name, (int)retorno.size, retorno.attrib );
        printf( "criacao: %d/%d/%d %d:%d:%d\n", horario.tm_mday,
                                       horario.tm_mon,
                                       horario.tm_year,
                                       horario.tm_hour,
                                       horario.tm_min,
                                       horario.tm_sec );

        localtime_s( &horario, &retorno.time_access );
        printf( "acesso: %d/%d/%d %d:%d:%d\n", horario.tm_mday,
                                       horario.tm_mon,
                                       horario.tm_year,
                                       horario.tm_hour,
                                       horario.tm_min,
                                       horario.tm_sec );

        localtime_s( &horario, &retorno.time_write );
        printf( "gravacao: %d/%d/%d %d:%d:%d\n", horario.tm_mday,
                                       horario.tm_mon,
                                       horario.tm_year,
                                       horario.tm_hour,
                                       horario.tm_min,
                                       horario.tm_sec );
        printf( "-------------\n" );
    } while ( _findnext( handle, &retorno ) == 0 );

//#define _A_NORMAL 0x00
//#define _A_RDONLY 0x01
//#define _A_HIDDEN 0x02
//#define _A_SYSTEM 0x04
//#define _A_SUBDIR 0x10
//#define _A_ARCH 0x20

    _findclose( handle );

    system ( "dir");
    printf( "\n\n\n" );

    return 0;
}
