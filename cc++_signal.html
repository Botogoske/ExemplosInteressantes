#include <stdio.h>
#include <signal.h>

void sinalizado( int p_sinal )
{
    switch( p_sinal )
    {
        case SIGINT:
            printf( "SIGINT\n" );
            break;
        case SIGILL:
            printf( "SIGILL\n" );
            break;
        case SIGABRT_COMPAT:
            printf( "SIGABRT_COMPAT\n" );
            break;
        case SIGFPE:
            printf( "SIGFPE\n" );
            break;
        case SIGSEGV:
            printf( "SIGSEGV\n" );
            break;
        case SIGTERM:
            printf( "SIGTERM\n" );
            break;
        case SIGBREAK:
            printf( "SIGBREAK\n" );
            break;
        case SIGABRT:
            printf( "SIGABRT\n" );
            break;
        default:
            printf( "indefinido.\n" );
            break;
    }
}

int main()
{
    void (*emitir_sinal)(int);

    emitir_sinal = signal( SIGINT, sinalizado ) ;
    emitir_sinal = signal( SIGTERM, sinalizado ) ;
    emitir_sinal = signal( SIGBREAK, sinalizado ) ;

    printf( "iniciando programa...\n" );
    raise( SIGINT );

    for( int i = 0; i < 100; ++i )
    {
        if ( i == 3 )
        {
            raise( SIGBREAK );
        }

        if ( i == 94 )
        {
            raise ( SIGTERM );
        }
    }

    printf( "fim.\n" );

    printf( "\n\n" );
    return 0;
}
