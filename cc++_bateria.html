#include <iostream>
#include <windows.h>

void bateria( void )
{
    std::cout << "BATERIA" << std::endl;
    std::cout << "-------" << std::endl;
    SYSTEM_POWER_STATUS * lps = new SYSTEM_POWER_STATUS;

    if ( ! GetSystemPowerStatus( lps ) )
    {
        std::cout << "bateria não disponível." << std::endl;
    }
    else if( lps->ACLineStatus == 1 )
    {
        std::cout << "bateria ligada na energia." << std::endl;
    }
    else
    {
        std::cout << "bateria em: " << lps->BatteryLifePercent << std::endl;
    }

    std::cout << std::endl;
}

int main()
{
    bateria();

    return 1;
}
