#include <iostream>
#include <windows.h>

void recursivo( std::string p_param )
{
    WIN32_FIND_DATAA data;

    std::string dir = p_param;
    dir.append( "*" );

    std::string imprimir = p_param;
    HANDLE hFind = FindFirstFileA( dir.c_str(), &data);

    if (hFind != INVALID_HANDLE_VALUE)
    {
        do
        {
            if ( data.dwFileAttributes == FILE_ATTRIBUTE_DIRECTORY )
            {
                if ( strcmp( data.cFileName, "." ) == 0 ||
                     strcmp( data.cFileName, ".." ) == 0 )
                {
                    continue;
                }

                dir = p_param;
                dir.append( data.cFileName );
                dir.append( "\\" );
                std::cout << dir << std::endl;
                recursivo( dir );
            }

        } while (FindNextFileA(hFind, &data));

        FindClose(hFind);
    }
}

int main()
{
    recursivo( "C:\\" );

    std::cout << std::endl;
    return 0;
}
