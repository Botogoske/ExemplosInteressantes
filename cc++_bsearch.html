#include <stdio.h>
#include <stdlib.h>

int comparar(const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}

int main ()
{
    int valor[ 6 ] = { 10, 20, 30, 40, 50, 60 };
    int chave = 30;
    int * achou = NULL;

     achou = (int*) bsearch (&chave, valor, 6, sizeof(int), comparar);


     if ( achou )
     {
         printf( "sim, achei: %d\n", *achou );
     }

  return 0;
}
