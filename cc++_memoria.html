#include <iostream>
#include <windows.h>

void memoria( void )
{
    std::cout << "MEMORIA" << std::endl;
    std::cout << "-------" << std::endl;

    MEMORYSTATUSEX statex;
    statex.dwLength = sizeof(statex);

    GlobalMemoryStatusEx(&statex);
    std::cout << "percent memory in use: " << statex.dwMemoryLoad << std::endl;

    std::cout << std::endl;
}

int main()
{
    memoria();

    return 1;
}
