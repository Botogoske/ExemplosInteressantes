#include <windows.h>
#include <stdio.h>
#include <tchar.h>

HWND g_button[ 2 ] = { 0, 0 };
HINSTANCE g_instance = 0;

LRESULT CALLBACK WindowProc(HWND hwnd, UINT uMsg, WPARAM wParam, LPARAM lParam)
{
    switch (uMsg)
    {
    case WM_SETFOCUS:
        printf( "WM_SETFOCUS \n" );
        break;

    case WM_ACTIVATE:
        printf( "WM_ACTIVATE \n" );
        break;

    case WM_CREATE:
        printf( "WM_CREATE \n" );
        break;

    case WM_SIZE:
        printf( "WM_SIZE \n" );
        break;

    case WM_MOVE:
//        printf( "WM_MOVE \n" );
        break;

    case WM_LBUTTONDOWN:
        printf( "WM_LBUTTONDOWN \n" );
        break;

    case WM_LBUTTONUP:
        printf( "WM_LBUTTONUP \n" );

        if ( g_button[ 0 ] == NULL )
        {
            g_button[ 0 ] = CreateWindow (
                TEXT("BUTTON"),                         // The class name required
                TEXT("Botão"),                     // the caption of the label
                WS_CHILD | WS_VISIBLE | BS_NOTIFY,  // the styles
                100, // the left co-ordinates x
                50,  // the top co-ordinates y
                80,  // width
                25,  // height
                hwnd, // parent window handle hwnd
                ( HMENU )NULL, // the ID of your button
                g_instance, // the instance of your application
                NULL)  // extrabits you dont really need
            ;
        }

        if ( g_button[ 1 ] == NULL )
        {
            g_button[ 1 ] = CreateWindow (
                TEXT("BUTTON"),                         // The class name required
                TEXT("Botão 2"),                     // the caption of the label
                WS_CHILD | WS_VISIBLE | BS_NOTIFY | BS_BITMAP,  // the styles
                100, // the left co-ordinates x
                100,  // the top co-ordinates y
                80,  // width
                25,  // height
                hwnd, // parent window handle hwnd
                ( HMENU )NULL, // the ID of your button
                g_instance, // the instance of your application
                NULL)  // extrabits you dont really need
            ;
        }
        break;

    case WM_COMMAND:
        if ( lParam == g_button[ 0 ] )
        {
            WORD acao = HIWORD( wParam );

            switch( acao )
            {
                case BN_CLICKED:
                    printf( "BN_CLICKED \n" );
                    break;

                case BN_DBLCLK:
                    printf( "BN_DBLCLK \n" );
                    break;

                case BN_SETFOCUS:
                    printf( "BN_SETFOCUS \n" );
                    break;

                case BN_KILLFOCUS:
                    printf( "BN_KILLFOCUS \n" );
                    break;

            }

        }
        break;

    case WM_RBUTTONDOWN:
        printf( "WM_RBUTTONDOWN \n" );
        break;

    case WM_RBUTTONUP:
        printf( "WM_RBUTTONUP \n" );
//        SetWindowText( g_button, TEXT( "Ola" ) );
//        EnableWindow( g_button, FALSE );
//        SetFocus( g_button[ 0 ] );
        HBITMAP hBMP = ( HBITMAP )LoadImage( NULL,
            TEXT( "logo16x16.bmp" ), IMAGE_BITMAP, 0, 0, LR_LOADFROMFILE );
        SendMessage( g_button[ 1 ], BM_SETIMAGE, (WPARAM)IMAGE_BITMAP, (LPARAM) hBMP );
        break;

        case WM_PAINT:
            {
                printf( "WM_PAINT \n" );

                PAINTSTRUCT ps;
                BeginPaint( hwnd, &ps );
                EndPaint( hwnd, &ps );
                return 0;
            }

        case WM_DESTROY:
            printf( "WM_DESTROY \n" );
            PostQuitMessage(0);
            return 0;
    }

    return DefWindowProc(hwnd, uMsg, wParam, lParam);
}


INT WINAPI WinMain(HINSTANCE hInstance, HINSTANCE hPrevInstance, PSTR lpCmdLine, INT nCmdShow)
{
    const char * CLASS_NAME  = TEXT("CLASSE_WINDOWS");
    g_instance = hInstance;

    WNDCLASS wc = { };

    wc.lpfnWndProc   = WindowProc;
    wc.hInstance     = hInstance;
    wc.lpszClassName = CLASS_NAME;

    RegisterClass(&wc);

     //Create the window.

    HWND hwnd = CreateWindowEx(
        0,
        CLASS_NAME,
        TEXT("Janela Windows"),
        WS_OVERLAPPEDWINDOW,
        CW_USEDEFAULT, // x position
        CW_USEDEFAULT, // y position
        CW_USEDEFAULT, // witdh
        CW_USEDEFAULT, // height
        NULL,
        NULL,
        hInstance,
        NULL
        );

    if (hwnd == NULL)
    {
        return 0;
    }

    ShowWindow(hwnd, nCmdShow);

     //Run the message loop.

    MSG msg = { };

    while (GetMessage(&msg, NULL, 0, 0))
    {
        TranslateMessage(&msg);
        DispatchMessage(&msg);
    }

    return 0;
}
