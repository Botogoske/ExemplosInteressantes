usuarios conectados


SELECT a.sid, 
       a.serial#, 
       a.status, 
       sql.disk_reads, 
       sql.direct_writes, 
       a.username, 
       a.machine, 
       a.program,sql.sql_text
  FROM v$session a, 
       v$sql sql
 WHERE sql.sql_id(+) = a.sql_id
   AND a.type = 'USER'
;
