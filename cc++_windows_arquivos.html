#include <iostream>
#include <string>
#include <windows.h>

int main()
{
    HANDLE arquivo = CreateFileA(
                "arquivo.txt",
                GENERIC_WRITE, //dwDesiredAccess,
                0, //FILE_SHARE_WRITE, // dwShareMode,
                NULL, // lpSecurityAttributes,
                CREATE_NEW, // dwCreationDisposition,
                FILE_ATTRIBUTE_NORMAL, // dwFlagsAndAttributes,
                NULL ) //hTemplateFile
            ;
            
    if ( arquivo == INVALID_HANDLE_VALUE )
    {
        std::cout << "erro na criacao do arquivo.";
        return 1;
    }
    
    std::string dados = "linha .......................\n" ;

    for( int i = 0; i < 10; ++i )
    {
        DWORD gravados = 0;

        WINBOOL rc = WriteFile( arquivo,
           dados.c_str(),
           dados.size(),
           &gravados,
           NULL
           );

        if ( gravados != dados.size() )
        {
            std::cout << "erro na gravacao do arquivo." << GetLastError() << std::endl;
            return 2;
        }
    }

    CloseHandle( arquivo );
    
    std::cout << "fim" << std::endl;

    return 0;
}

#include <iostream>
#include <string>
#include <windows.h>

int main()
{
    HANDLE arquivo = CreateFileA(
                "arquivo.txt",
                GENERIC_READ, //dwDesiredAccess,
                0, //FILE_SHARE_WRITE, // dwShareMode,
                NULL, // lpSecurityAttributes,
                OPEN_EXISTING, // dwCreationDisposition,
                FILE_ATTRIBUTE_NORMAL, // dwFlagsAndAttributes,
                NULL ) //hTemplateFile
            ;

    if ( arquivo == INVALID_HANDLE_VALUE )
    {
        std::cout << "erro na leitura do arquivo.";
        return 1;
    }

    DWORD lidos = 0;
    char byte;
    std::string dados;

    while( true )
    {
        WINBOOL rc = ReadFile( arquivo,
           &byte,
           sizeof( byte ),
           &lidos,
           NULL
           );

        if ( rc && lidos == 0 )
        {
            break;
        }

        if ( byte == '\n' )
        {
            std::cout << dados << std::endl;
            dados.clear();
            continue;
        }

        dados += byte;
    }

    std::cout << dados << std::endl;

    CloseHandle( arquivo );

    return 0;
}

#include <iostream>
#include <string>
#include <windows.h>

int main()
{
    struct  _dados
    {
            char nome[ 30 + 1 ];
            int idade;
    };

    struct _dados dados;
    memset( dados.nome, '\0', sizeof( dados.nome ));
    strncpy( dados.nome, "String de dados.", 16 );
    dados.idade = 22;
    
    HANDLE arquivo = CreateFileA(
                "estrutura.txt",
                GENERIC_WRITE,
                0,
                NULL,
                CREATE_NEW,
                FILE_ATTRIBUTE_NORMAL,
                NULL );

    if ( arquivo == INVALID_HANDLE_VALUE )
    {
        std::cout << "erro na criacao do arquivo" << std::endl;
        return 1;
    }

    DWORD gravados = 0;

    WINBOOL rc = WriteFile( arquivo,
                            &dados,
                            sizeof( dados ),
                            &gravados,
                            NULL );

    if ( gravados != sizeof( dados ) )
    {
        std::cout << "erro na gravacao do arquivo" << std::endl;
        return 2;

    }

    std::cout << "nome: " << dados.nome << std::endl;
    std::cout << "idade: " << dados.idade << std::endl;

    CloseHandle( arquivo );

    std::cout << "fim" << std::endl;
    return 0;
}
