empregado.h

#ifndef EMPREGADO_H
#define EMPREGADO_H
#include <string>

class Empregado
{
    public:
        Empregado();
        virtual void setNome( char * );
        virtual void setProfissao( char * );
        virtual void setSexo( char ) = 0;
        virtual char getSexo( void ) = 0;

        std::string getNome( void ) { return this->nome; }
        std::string getProfissao( void ) { return this->profissao; }

    protected:
        std::string nome;
        std::string profissao;
};

#endif // EMPREGADO_H
