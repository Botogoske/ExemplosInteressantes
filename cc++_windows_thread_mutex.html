#include <iostream>
#include <windows.h>

CRITICAL_SECTION lock;

std::string GetLastErrorAsString()
{
    DWORD errorMessageID = ::GetLastError();

    if ( errorMessageID == 0 )
    {
        return std::string();
    }

    LPSTR messageBuffer = nullptr;

    size_t size = FormatMessageA(
        FORMAT_MESSAGE_ALLOCATE_BUFFER |
        FORMAT_MESSAGE_FROM_SYSTEM |
        FORMAT_MESSAGE_IGNORE_INSERTS,
        NULL,
        errorMessageID,
        MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),
        (LPSTR)&messageBuffer,
        0,
        NULL);

    std::string message(messageBuffer, size);

    LocalFree(messageBuffer);

    return message;
}

DWORD WINAPI Thread_no_1( LPVOID lpParam )
{
    int * valor = ( int * )lpParam;
    srand( 1 );

    for (int i = 0; i < 10; i++ )
    {
        EnterCriticalSection( &lock );
        int tmp = *valor;
        tmp++;
        *valor = tmp;
        std::cout << "thread 1   " << *valor << std::endl;
        LeaveCriticalSection( &lock );
        Sleep( rand() % 1500 );
    }

    return 0;
}

DWORD WINAPI Thread_no_2( LPVOID lpParam )
{
    int * valor = ( int * )lpParam;
    srand( 1 );

    for (int i = 0; i < 10; i++ )
    {
        EnterCriticalSection( &lock );
        int tmp = *valor;
        tmp++;
        *valor = tmp;
        LeaveCriticalSection( &lock );
        std::cout << "thread 2   " << tmp << std::endl;
        Sleep( rand() % 500 );
    }

    return 0;
}

int main()
{
    int Data_Of_Thread = 0;
    HANDLE thread[ 2 ];

    InitializeCriticalSection( &lock );

    thread[ 0 ] = CreateThread(
                NULL,
                0,
                Thread_no_1,
                &Data_Of_Thread,
                0,
                NULL);

    if ( !thread[ 0 ] )
    {
        return 1;
    }

    thread[ 1 ] = CreateThread(
                NULL,
                0,
                Thread_no_2,
                &Data_Of_Thread,
                0,
                NULL);

    if ( !thread[ 1 ] )
    {
        return 2;
    }

    DWORD rc = WaitForMultipleObjects( 2, thread, TRUE, INFINITE );

    switch( rc )
    {
        case WAIT_ABANDONED:
            std::cout << "Mutex n√£o liberado pela thread" << std::endl;
            break;
        case WAIT_OBJECT_0:
            std::cout << "Estado do objecto sinalizado." << std::endl;
            break;
        case WAIT_TIMEOUT:
            std::cout << "Tempo esgotado" << std::endl;
            break;
        case WAIT_FAILED:
            std::cout << "Falha de execucao: " << GetLastErrorAsString() << std::endl;
            break;
        default:
            std::cout << "indefinido: " << GetLastErrorAsString() << std::endl;
            break;
    }

    CloseHandle( thread );

    DeleteCriticalSection( &lock );
    std::cout << std::endl;
    return 1;
}
