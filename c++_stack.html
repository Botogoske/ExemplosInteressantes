Vídeo do Canal 

#include <iostream>
#include <stack>

int main()
{
    struct _dados
    {
        std::string nome;
        int idade;
    };

    std::stack< struct _dados > fila;

    struct _dados dados;

    dados.nome = "Wagner";
    dados.idade = 22;
    fila.push( dados );

    dados.nome = "Esposa";
    dados.idade = 33;
    fila.push( dados );

    dados.nome = "Filho";
    dados.idade = 44;
    fila.push( dados );

    std::cout << "tamanho: " << fila.size() << std::endl;

    while( !fila.empty() )
    {
        struct _dados tmp = fila.top();

        std::cout << "nome: " << tmp.nome << std::endl;
        std::cout << "idade: " << tmp.idade << std::endl;

        fila.pop();
    }

    std::cout << std::endl << std::endl;
    return 0;
}
